\subsection{Preference Engineering}
\frameT{Preference Learning}{
	Goal: preference predicting and feasibility of various preference models.
	\begin{enumerate}
		\item Recommender Systems:
			\begin{enumerate}
				\item Collaborative
				\item Content-based
				\item Hybrid
			\end{enumerate}
		\item Machine Learning:
			\begin{enumerate}
				\item Supervised learning
				\item Label ranking
			\end{enumerate}
		\item Preference Elicitation:
			\begin{enumerate}
				\item Positive learning
				\item Human-in-the-loop
			\end{enumerate}
		\item Preference Learning:
			\begin{enumerate}
				\item Conditional Preference Networks
				\item Preference Trees (e.g., LP-trees, CLP-trees, PLP-trees)
				\item Stochastic Models (e.g., Choquet integral, TOPSIS-like models)
			\end{enumerate}
	\end{enumerate}
}

\frameT{Preference Reasoning}{
	Goal: personalized optimization and collaborative decision making.

	\begin{enumerate}
		\item Preference Optimization:
			\begin{enumerate}
				\item Consistency testing
				\item Dominance testing
				\item Optimality testing
			\end{enumerate}
		\item Preference Aggregation:
			\begin{enumerate}
				\item Social choice and welfare
				\item Voting systems
			\end{enumerate}
		\item Preference Misrepresentation:
			\begin{enumerate}
				\item Control
				\item Manipulation
				\item Bribery
			\end{enumerate}
	\end{enumerate}
}

\frameT{Preference Applications}{
	Goal: exploring possibilities of collaboration with experts in other areas.

	\begin{enumerate}
		\item Automated Planning and Scheduling:
			\begin{enumerate}
				\item Travel scheduling
				\item Manufacturing
				\item Traffic control
			\end{enumerate}
		\item Computer Vision and Image Processing:
			\begin{enumerate}
				\item Image retrieval
				\item Image and video understanding
			\end{enumerate}
	\end{enumerate}
}
