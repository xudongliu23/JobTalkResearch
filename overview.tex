	Q: How do we represent preferences over combinatorial domains?

	\begin{enumerate}
		\item Quantitative:
			\begin{enumerate}
				\item Utility/Cost Functions
				\item Possibilistic Logic\footcitefull{DuboisLP91}
				\item Fuzzy Preference Relations\footcitefull{orlovsky1978decision}
				\item Penalty Logic\footcitefull{pinkas1991propositional}
			\end{enumerate}
		\seti
	\end{enumerate}
}

\frameT{Research on Preferences}{
	Q: How do we learn/predict preferences over combinatorial domains?

	\begin{enumerate}
		\item Recommender Systems\footcitefull{adomavicius2005toward}:
			\begin{enumerate}
				\item Collaborative
				\item Content-based
				\item Hybrid
			\end{enumerate}
		\item Machine Learning:
			\begin{enumerate}
				\item Supervised learning
				\item Label ranking
			\end{enumerate}
		\item Preference Elicitation:
			\begin{enumerate}
				\item Positive learning
				\item Human-in-the-loop
			\end{enumerate}
		\item Preference Learning\footcitefull{furnkranz2010preference}:
			\begin{enumerate}
				\item Conditional Preference Networks
				\item Preference Trees (e.g., LP-trees, CLP-trees, PLP-trees)
			\end{enumerate}
	\end{enumerate}

}
