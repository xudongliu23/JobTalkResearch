\frameT{Research on Preferences}{
	Q: How do we represent preferences over combinatorial domains?

	\begin{enumerate}
		\item Quantitative:
			\begin{enumerate}
				\item Utility/Cost Functions
				\item Possibilistic Logic\cite{DuboisLP91}
				\item Fuzzy Preference Relations\cite{orlovsky1978decision}
				\item Penalty Logic\cite{pinkas1991propositional}
			\end{enumerate}
		\item Qualitative:
			\begin{enumerate}
				\item Preference Trees (e.g., 
							LP-trees\cite{booth:learningLP,conf/adt13/LiuT}, 
							CLP-trees\cite{brauning2012learning}, 
							PLP-trees\cite{conf/aaai15/LiuT}, P-trees\cite{fraser1994,wsh/mpref14/LiuT})
				\item Conditional Preference Networks\cite{bbdh03}
				\item Answer-Set Optimization Theories\cite{Brewka:ASO}
			\end{enumerate}
		\item Hybrid:
			\begin{enumerate}
				\item Preference Forests
			\end{enumerate}
	\end{enumerate}
}

\frameT{Research on Preferences}{
	Q: How do we learn/predict preferences over combinatorial domains?

	\begin{enumerate}
		\item Recommender Systems\cite{adomavicius2005toward}:
			\begin{enumerate}
				\item Collaborative
				\item Content-based
				\item Hybrid
			\end{enumerate}
		\item Machine Learning:
			\begin{enumerate}
				\item Supervised learning
				\item Label ranking
			\end{enumerate}
		\item Preference Elicitation:
			\begin{enumerate}
				\item Positive learning
				\item Human-in-the-loop
			\end{enumerate}
		\item Preference Learning\cite{furnkranz2010preference}:
			\begin{enumerate}
				\item Conditional Preference Networks
				\item Preference Trees (e.g., LP-trees, CLP-trees, PLP-trees)
			\end{enumerate}
	\end{enumerate}

}
