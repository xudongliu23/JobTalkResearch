\frameT{Data-Driven Preference Engineering}{
	\begin{enumerate}
		\item Goal: fitting functions vs. learning interpretable decision models.
		\item Data: binary vs. graded, absolute vs. relative, explicit vs. implicit,
					structured vs. unstructured, and single vs. multiple users.
			\begin{itemize}
				\item Building datasets (Data mining and filtering techiques, with tools such as
							Python, Matlab).
			\end{itemize}
		\item Methods/Models: recommender systems, supervised learning, conditional preference
					networks, and preference trees.
			\begin{itemize}
				\item Reading related literature and implementing systems (c/c++, logic programming, Matlab, etc).
			\end{itemize}
		\seti
	\end{enumerate}
}

\frameT{Data-Driven Preference Engineering}{
	\begin{enumerate}
		\conti
		\item Reasoning tasks: dominance and optimization, and preference aggregation.
			\begin{itemize}
				\item Implementing decision support tools.
			\end{itemize}
		\item Applications: fair division, strategyproof social science, and planning and scheduling.
			\begin{itemize}
				\item Proving computational complexities, and design and implementing approximate algorithms.
			\end{itemize}
	\end{enumerate}
